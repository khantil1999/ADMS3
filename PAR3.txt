DECLARE 
 x NUMBER;
 EMP_COUNT NUMBER;
 SALARY NUMBER;
 HRA NUMBER;
 DA NUMBER;
 NET_SALARY NUMBER; 
BEGIN 
x:=:ENTER_EMPLOYEE_NUMBER;
BEGIN
SELECT COUNT(*) INTO "EMP_COUNT" FROM EMPLOYEE1 WHERE ID=x;
DBMS_OUTPUT.PUT_LINE(EMP_COUNT); 
BEGIN 
IF EMP_COUNT = 1 THEN
  SELECT SALARY INTO "SALARY" FROM EMPLOYEE1 WHERE ID=x;
  DBMS_OUTPUT.PUT_LINE(SALARY);
  HRA:=(SALARY*10)/100;
  DA:=(SALARY*60)/100;
  NET_SALARY:=SALARY+HRA+DA;
  DBMS_OUTPUT.PUT_LINE('YOUR BASIC SALARY IS:' || NET_SALARY);
ELSE
 DBMS_OUTPUT.PUT_LINE('SORRY NO EMPLOYEE FOUND'); 
END IF;
END;
END;
END;

DECLARE 
 x NUMBER;
 EMP_COUNT NUMBER;
 SALARY NUMBER;
  
BEGIN 
x:=:ENTER_EMPLOYEE_NUMBER;
BEGIN
SELECT COUNT(*) INTO "EMP_COUNT" FROM EMPLOYEE1 WHERE ID=x;

BEGIN 
IF EMP_COUNT = 1 THEN
  SELECT SALARY INTO "SALARY" FROM EMPLOYEE1 WHERE ID=x;
  BEGIN
     IF SALARY < 10000 THEN
        DBMS_OUTPUT.PUT_LINE('LESS SALARY');

     ELSIF SALARY >10000 AND SALARY < 30000 THEN
         DBMS_OUTPUT.PUT_LINE('MEDIUM SALARY');

     ELSIF SALARY >30000 AND SALARY < 60000 THEN
         DBMS_OUTPUT.PUT_LINE('HIGH SALARY');

     ELSIF SALARY >60000 THEN
         DBMS_OUTPUT.PUT_LINE('VERY HIGH SALARY');
     END IF;
   END;
ELSE
 DBMS_OUTPUT.PUT_LINE('SORRY NO EMPLOYEE FOUND'); 
END IF;
END;
END;
END;
